    if (start >= end) {
        return;

    // 左右指针
    int left = start;
    int right = end;
    int poivt = data[start]; // 选择第一个数 作为锚点
    while (left < right) {
        // 从右边找一个比锚点poivt更小的值 data[right]

        }
