class Solution {
public:
    int tribonacci(int n) {
        if (n <= 2) {
            return n == 0 ? 0 : 1;
        }
        vector<int> f(n+1);
        f[0] = 0;
        f[1] = f[2] = 1;
        for (int i = 3; i < f.size(); ++i) {
            f[i] = f[i-1] + f[i-2] + f[i-3];
        }
        return f.back();
    }
};
